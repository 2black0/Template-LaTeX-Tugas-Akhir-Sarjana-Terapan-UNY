% Macro to resize both name and NIP if name would wrap to two lines
\newcommand{\smartsignature}[2]{%
    % Test if name fits in one line at normal size
    \setbox0=\hbox{#1}%
    \ifdim\wd0>0.47\textwidth
        % Name is too wide, use smaller font for both name and NIP
        {\small #1\\[0.3em]#2}%
    \else
        % Name fits, use normal size for both
        {#1\\[0.3em]#2}%
    \fi
}
